        function removeURLParameter(url, parameter) {
            //prefer to use l.search if you have a location/link object
            var urlparts = url.split('?');
            if (urlparts.length >= 2) {

                var prefix = encodeURIComponent(parameter) + '=';
                var pars = urlparts[1].split(/[&;]/g);

                //reverse iteration as may be destructive
                for (var i = pars.length; i-- > 0;) {
                    //idiom for string.startsWith

                        pars.splice(i, 1);
                    }
                }

                return urlparts[0] + (pars.length > 0 ? '?' + pars.join('&') : '');
            }

        }

    setTimeout(function() {
        if (typeof jQuery === 'undefined') return;
        
        jQuery(document).ready(function($) {
            var $burger = $('.cs-header__offcanvas-toggle');
            var $offcanvas = $('.cs-offcanvas');
            var $overlay = $('.offcanvas-overlay');
            var $closeBtn = $offcanvas.find('.cs-offcanvas__toggle, .cs-offcanvas__close, .close');
            
            if ($burger.length && $offcanvas.length) {
                $offcanvas.removeClass('active show open').css('left', '-100%');
                
                $burger.on('click', function(e) {
                    e.preventDefault();
                    $offcanvas.addClass('active').css('left', '0');
                    $overlay.addClass('active');
                    $('body').addClass('offcanvas-open');
                });
                
                $closeBtn.on('click', function(e) {
                    e.preventDefault();
                    $offcanvas.removeClass('active').css('left', '-100%');
                    $overlay.removeClass('active');
                    $('body').removeClass('offcanvas-open');
                });
                
                $overlay.on('click', function() {
                    $offcanvas.removeClass('active').css('left', '-100%');
                    $overlay.removeClass('active');
                    $('body').removeClass('offcanvas-open');
                });
                
                $(document).on('keyup', function(e) {
                    if (e.key === 'Escape' && $offcanvas.hasClass('active')) {
                        $offcanvas.removeClass('active').css('left', '-100%');
                        $overlay.removeClass('active');
                        $('body').removeClass('offcanvas-open');
                    }
                });
            }
        });
    }, 500);
});

    <script>
    document.addEventListener('DOMContentLoaded', function() {
        const toggle = document.querySelector('.cs-header__search-toggle');
        const searchBlock = document.querySelector('.cs-search');
        
        if (!toggle || !searchBlock) return;
        
        // Останавливаем jQuery анимации
        if (typeof jQuery !== 'undefined') {
            jQuery(searchBlock).stop(true, true);
        }
        
        // Создаем новый обработчик
        const newToggle = toggle.cloneNode(true);
        toggle.parentNode.replaceChild(newToggle, toggle);
        
        // Стили
        searchBlock.style.display = '';
        
        let isOpen = false;
        
        newToggle.addEventListener('click', function(e) {
            e.preventDefault();
            
            if (isOpen) {
                searchBlock.style.display = 'none';
            } else {
                searchBlock.style.display = 'block';
                searchBlock.style.height = 'auto';
                setTimeout(() => {
                    const input = searchBlock.querySelector('input[type="search"], input[name="s"]');
                    if (input) input.focus();
                }, 50);
            }
            
            isOpen = !isOpen;
        });
    });
    </script>
